//bubble sort to sort in ascending order on roll number


void bubble_sort(stud s[20],int n)
{
      int i,j;
      stud temp;
      for(i=1;i<n;i++)
      {
                 for(j=0;j<n-i;j++)
                 {
                       if(s[j].roll_num>s[j+1].roll_num)
                       {
                              temp=s[j];
                              s[j]=s[j+1];
                              s[j+1]=temp;
                       }
                 }
      }
}
